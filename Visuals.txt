--[[
    VISUALS v5
    
    Speed Hack: WalkSpeed override
    Viewmodel Chams: Highlight on Camera.FirstPersonRig
    Crosshair: Cross / Dot / Circle / T
    World Colors: Saturation / Contrast / Brightness
    
    Control via getgenv().Visuals.Settings
]]

local Players    = game:GetService("Players")
local RunService = game:GetService("RunService")
local UIS        = game:GetService("UserInputService")
local Lighting   = game:GetService("Lighting")
local Camera     = workspace.CurrentCamera
local LP         = Players.LocalPlayer

-- ═══════════════════════════════════════════
-- SETTINGS
-- ═══════════════════════════════════════════
local S = {
    Enabled = true,

    -- Speed Hack
    SpeedHack    = false,
    WalkSpeed    = 25,   -- default game speed is 16
    BaseSpeed    = 16,

    -- Viewmodel Chams (Camera > FirstPersonRig)
    VMChams          = false,
    VMChamsColor     = Color3.fromRGB(108, 92, 231),
    VMChamsFill      = 0.5,   -- fill transparency (0 = solid, 1 = invisible)
    VMChamsOutline   = 0,     -- outline transparency
    VMChamsOutColor  = Color3.fromRGB(255, 255, 255),
    VMChamsGunOnly   = false, -- only highlight the gun, not arms

    -- World Colors (ColorCorrectionEffect)
    WorldColors  = false,
    Saturation   = 0,
    Contrast     = 0,
    Brightness   = 0,

    -- Crosshair
    Crosshair        = true,
    CrosshairMode    = "Cross",
    CrosshairColor   = Color3.fromRGB(0, 255, 0),
    CrosshairSize    = 10,
    CrosshairThick   = 1.5,
    CrosshairGap     = 4,
    CrosshairDot     = true,
    CrosshairOutline = true,
}

-- ═══════════════════════════════════════════
-- VIEWMODEL CHAMS
-- Targets: workspace.CurrentCamera.FirstPersonRig
-- Contains: gun model (M870, AK-47, etc), Head, Arms
-- We put a Highlight on the whole rig or just the gun
-- ═══════════════════════════════════════════
local VMHighlight = nil      -- Highlight on full rig
local VMGunHighlight = nil   -- Highlight on gun only (when GunOnly mode)
local LastRigRef = nil

local function GetFirstPersonRig()
    Camera = workspace.CurrentCamera
    return Camera and Camera:FindFirstChild("FirstPersonRig")
end

-- Find the gun model inside FirstPersonRig
-- It's the child that ISN'T a body part, Humanoid, Shirt, Pants, etc
local BodyPartNames = {
    Head = true, ["Left Arm"] = true, ["Right Arm"] = true,
    ["Left Leg"] = true, ["Right Leg"] = true, Torso = true,
    HumanoidRootPart = true, Humanoid = true, Shirt = true,
    Pants = true, ["Body Colors"] = true, Animate = true,
    -- R15
    UpperTorso = true, LowerTorso = true, LeftUpperArm = true,
    LeftLowerArm = true, LeftHand = true, RightUpperArm = true,
    RightLowerArm = true, RightHand = true,
}

local function FindGunModel(rig)
    if not rig then return nil end
    for _, child in rig:GetChildren() do
        if child:IsA("Model") or child:IsA("Tool") or child:IsA("Accessory") then
            return child
        end
        -- Also check: anything that's not a known body part and has descendants (likely the gun)
        if not BodyPartNames[child.Name] and not child:IsA("Humanoid") and not child:IsA("Shirt") and not child:IsA("Pants") and not child:IsA("BodyColors") then
            if child:IsA("BasePart") or child:IsA("MeshPart") or (#child:GetChildren() > 0) then
                return child
            end
        end
    end
    return nil
end

local function CleanVMChams()
    if VMHighlight then pcall(function() VMHighlight:Destroy() end); VMHighlight = nil end
    if VMGunHighlight then pcall(function() VMGunHighlight:Destroy() end); VMGunHighlight = nil end
    LastRigRef = nil
end

local function UpdateVMChams()
    if not S.VMChams then
        CleanVMChams()
        return
    end

    local rig = GetFirstPersonRig()
    if not rig then
        CleanVMChams()
        return
    end

    if S.VMChamsGunOnly then
        -- Gun-only mode: highlight just the weapon
        if VMHighlight then pcall(function() VMHighlight:Destroy() end); VMHighlight = nil end

        local gun = FindGunModel(rig)
        if not gun then
            if VMGunHighlight then pcall(function() VMGunHighlight:Destroy() end); VMGunHighlight = nil end
            return
        end

        if not VMGunHighlight or VMGunHighlight.Parent ~= gun then
            if VMGunHighlight then pcall(function() VMGunHighlight:Destroy() end) end
            VMGunHighlight = Instance.new("Highlight")
            VMGunHighlight.Name = "VIS_VMGun"
            VMGunHighlight.DepthMode = Enum.HighlightDepthMode.AlwaysOnTop
            VMGunHighlight.Adornee = gun
            VMGunHighlight.Parent = gun
        end

        VMGunHighlight.FillColor = S.VMChamsColor
        VMGunHighlight.FillTransparency = S.VMChamsFill
        VMGunHighlight.OutlineColor = S.VMChamsOutColor
        VMGunHighlight.OutlineTransparency = S.VMChamsOutline
    else
        -- Full rig mode: highlight everything (gun + arms)
        if VMGunHighlight then pcall(function() VMGunHighlight:Destroy() end); VMGunHighlight = nil end

        if not VMHighlight or VMHighlight.Parent ~= rig or LastRigRef ~= rig then
            if VMHighlight then pcall(function() VMHighlight:Destroy() end) end
            VMHighlight = Instance.new("Highlight")
            VMHighlight.Name = "VIS_VM"
            VMHighlight.DepthMode = Enum.HighlightDepthMode.AlwaysOnTop
            VMHighlight.Adornee = rig
            VMHighlight.Parent = rig
            LastRigRef = rig
        end

        VMHighlight.FillColor = S.VMChamsColor
        VMHighlight.FillTransparency = S.VMChamsFill
        VMHighlight.OutlineColor = S.VMChamsOutColor
        VMHighlight.OutlineTransparency = S.VMChamsOutline
    end
end

-- ═══════════════════════════════════════════
-- SPEED HACK
-- ═══════════════════════════════════════════
local function UpdateSpeed()
    if not S.SpeedHack then return end
    local ch = LP.Character
    if not ch then return end
    local hum = ch:FindFirstChildOfClass("Humanoid")
    if hum then
        hum.WalkSpeed = S.WalkSpeed
    end
end

local function RestoreSpeed()
    local ch = LP.Character
    if not ch then return end
    local hum = ch:FindFirstChildOfClass("Humanoid")
    if hum then
        hum.WalkSpeed = S.BaseSpeed
    end
end

-- ═══════════════════════════════════════════
-- COLOR CORRECTION
-- ═══════════════════════════════════════════
local CC = nil

local function GetCC()
    if CC and CC.Parent then return CC end
    for _, obj in Lighting:GetChildren() do
        if obj.Name == "VIS_CC" then obj:Destroy() end
    end
    CC = Instance.new("ColorCorrectionEffect")
    CC.Name = "VIS_CC"
    CC.Parent = Lighting
    return CC
end

local function UpdateCC()
    if S.WorldColors then
        local cc = GetCC()
        cc.Brightness = S.Brightness
        cc.Saturation = S.Saturation
        cc.Contrast   = S.Contrast
        cc.TintColor  = Color3.fromRGB(255, 255, 255)
        cc.Enabled    = true
    else
        if CC and CC.Parent then CC.Enabled = false end
    end
end

-- ═══════════════════════════════════════════
-- CROSSHAIR (Drawing)
-- ═══════════════════════════════════════════
local CL, CO = {}, {}
for i = 1, 4 do
    local l = Drawing.new("Line"); l.Visible = false; l.Transparency = 1; l.ZIndex = 100; CL[i] = l
    local o = Drawing.new("Line"); o.Visible = false; o.Color = Color3.new(0,0,0); o.Transparency = 0.8; o.ZIndex = 99; CO[i] = o
end
local CDot = Drawing.new("Circle"); CDot.Visible = false; CDot.Filled = true; CDot.NumSides = 16; CDot.ZIndex = 101
local CDotOut = Drawing.new("Circle"); CDotOut.Visible = false; CDotOut.Filled = false; CDotOut.Color = Color3.new(0,0,0); CDotOut.Thickness = 2; CDotOut.NumSides = 16; CDotOut.ZIndex = 100
local CCirc = Drawing.new("Circle"); CCirc.Visible = false; CCirc.Filled = false; CCirc.NumSides = 40; CCirc.ZIndex = 100
local CCircOut = Drawing.new("Circle"); CCircOut.Visible = false; CCircOut.Filled = false; CCircOut.Color = Color3.new(0,0,0); CCircOut.NumSides = 40; CCircOut.ZIndex = 99

local function HideCH()
    for i = 1, 4 do CL[i].Visible = false; CO[i].Visible = false end
    CDot.Visible = false; CDotOut.Visible = false; CCirc.Visible = false; CCircOut.Visible = false
end

local function SetLine(idx, x1, y1, x2, y2)
    CL[idx].From = Vector2.new(x1, y1); CL[idx].To = Vector2.new(x2, y2)
    CL[idx].Color = S.CrosshairColor; CL[idx].Thickness = S.CrosshairThick; CL[idx].Visible = true
    if S.CrosshairOutline then
        CO[idx].From = CL[idx].From; CO[idx].To = CL[idx].To
        CO[idx].Thickness = S.CrosshairThick + 2; CO[idx].Visible = true
    end
end

local function RenderCH()
    if not S.Crosshair then HideCH(); return end
    HideCH()
    Camera = workspace.CurrentCamera
    local vp = Camera.ViewportSize
    local cx, cy = math.floor(vp.X / 2), math.floor(vp.Y / 2)
    local sz, gp = S.CrosshairSize, S.CrosshairGap
    local mode = S.CrosshairMode

    if mode == "Cross" then
        SetLine(1, cx, cy-gp-sz, cx, cy-gp)
        SetLine(2, cx, cy+gp, cx, cy+gp+sz)
        SetLine(3, cx-gp-sz, cy, cx-gp, cy)
        SetLine(4, cx+gp, cy, cx+gp+sz, cy)
    elseif mode == "T" then
        SetLine(1, cx, cy+gp, cx, cy+gp+sz)
        SetLine(2, cx-gp-sz, cy, cx-gp, cy)
        SetLine(3, cx+gp, cy, cx+gp+sz, cy)
    elseif mode == "Circle" then
        CCirc.Position = Vector2.new(cx, cy); CCirc.Radius = sz
        CCirc.Color = S.CrosshairColor; CCirc.Thickness = S.CrosshairThick; CCirc.Visible = true
        if S.CrosshairOutline then
            CCircOut.Position = Vector2.new(cx, cy); CCircOut.Radius = sz; CCircOut.Thickness = S.CrosshairThick + 2; CCircOut.Visible = true
        end
    end

    if S.CrosshairDot or mode == "Dot" then
        local r = mode == "Dot" and math.max(S.CrosshairThick + 1, 3) or S.CrosshairThick
        CDot.Position = Vector2.new(cx, cy); CDot.Radius = r; CDot.Color = S.CrosshairColor; CDot.Visible = true
        if S.CrosshairOutline then CDotOut.Position = Vector2.new(cx, cy); CDotOut.Radius = r + 1; CDotOut.Visible = true end
    end
end

-- ═══════════════════════════════════════════
-- MAIN LOOP
-- ═══════════════════════════════════════════
local RENDER = "VIS_v5"
pcall(function() RunService:UnbindFromRenderStep(RENDER) end)

RunService:BindToRenderStep(RENDER, Enum.RenderPriority.Camera.Value + 1, function()
    if not S.Enabled then return end
    Camera = workspace.CurrentCamera

    UpdateSpeed()
    UpdateVMChams()
    UpdateCC()
    RenderCH()
end)

-- ═══════════════════════════════════════════
-- RESTORE / CLEANUP
-- ═══════════════════════════════════════════
local function RestoreAll()
    RestoreSpeed()
    CleanVMChams()
    if CC and CC.Parent then CC:Destroy() end; CC = nil
    HideCH()
end

local function Cleanup()
    pcall(function() RunService:UnbindFromRenderStep(RENDER) end)
    RestoreAll()
    for i = 1, 4 do CL[i]:Remove(); CO[i]:Remove() end
    CDot:Remove(); CDotOut:Remove(); CCirc:Remove(); CCircOut:Remove()
end

-- ═══════════════════════════════════════════
-- API
-- ═══════════════════════════════════════════
getgenv().Visuals = {
    Settings = S,
    Cleanup  = Cleanup,
    Restore  = RestoreAll,
    Toggle   = function() S.Enabled = not S.Enabled; if not S.Enabled then RestoreAll() end end,
}
